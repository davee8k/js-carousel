jQuery Carousel

Basic settings

$("#carousel-1").carousel({});

Custom move and speed

$("#carousel-2").carousel({move: "100%", speed: 0.3});

Endless with custom element

$("#carousel-3").carousel({item: "span", endless: true});

With pagination

$("#carousel-4").carousel({pager: true, pagerClass: "text-center", move: 3, perPage: 3});

CSS animation

$("#carousel-5").carousel({cssOnly: true});

.carousel-content { transition: left 0.3s linear; }

Automatic scroll with Timer

Require Timer (not included in carousel.js source)

$("#carousel-6").carousel({endless: true, pause: 1000}); 

// universal timer
function Timer(t,i){var n,e,o=!1,u=t,s=i;this.pause=function(){o=!0,n&&window.clearTimeout(n),s-=new Date-e},this.restart=function(t){void 0!==t&&(u=t),n&&window.clearTimeout(n),s=i,this.play()},this.play=function(){o=!1,e=new Date,n=window.setTimeout(u,s)},this.getTime=function(){return s},this.isPaused=function(){return o},this.play()};

Multiple rows

$("#carousel-7").carousel({rows: 2});

Vertical carousel

$("#carousel-8").carousel({direction: "vertical", endless: true});
